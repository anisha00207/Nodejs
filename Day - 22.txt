2ï¸âƒ£ Route Guards (Protecting Routes)

import { CanActivate, Router } from '@angular/router';

@Injectable({
  providedIn: 'root'
})
export class AuthGuard implements CanActivate {
  constructor(private router: Router) {}

  canActivate(): boolean {
    const isAuthenticated = false; // Simulating authentication check
    if (!isAuthenticated) {

      return false;
    }
    return true;
  }
}

ğŸ”¹ Route Guards (Protect Routes)

ğŸ“Œ Create an Auth Guard (auth.guard.ts):

    }
    return true;
  }
